PROJECT(dataoutput)

                ./dataoutput1.cpp)

                ./dataoutput_test.cpp)

INCLUDE_DIRECTORIES(${EIGEN_INCLUDE_DIRS})

SET(CMAKE_CXX_FLAGS_DEBUG -g)

ADD_LIBRARY(${PROJECT_NAME} STATIC ${DO_LIB_SRC})
ADD_EXECUTABLE(dop_test ${DO_LIB_TEST_SRC})

#Settings for test
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/../datareader)
TARGET_LINK_LIBRARIES(dop_test datareader)
